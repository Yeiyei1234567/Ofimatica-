\item $\displaystyle \frac{d}{dx}(c)=0.$
\item $\displaystyle \frac{d}{dx}(x)=1.$
\item $\displaystyle \frac{d}{dx}(u+v-w)=\frac{du}{dx}+\frac{dv}{dx}-\frac{dw}{dx}.$
\item $\displaystyle \frac{d}{dx}(cu)=c \cdot \frac{du}{dx}.$
\item $\displaystyle \frac{d}{dx}(uv)=u \cdot \frac{dv}{dx}+v \cdot \frac{du}{dx}.$
\item $\displaystyle \frac{d}{dx}(u^n)=nu^{n-1} \frac{du}{dx}.$
\item $\displaystyle \frac{d}{dx}  \left( \frac{u}{v}  \right =  \frac{v \cdot \frac{du}{dx}-u  \cdot  \frac{dv}{dx}}{v^2}.$
\item $\displaystyle \frac{d}{dx} \left( \frac{u}{v} \right) = \frac{1}{c} \cdot \frac{du}{dx}.$
\item $\displaystyle \frac{d}{dx} = \frac{dy}{dv} \cdot \frac{dv}{dx}.$
\item $\displaystyle \frac{d}{dx}(y) = \frac{1}{\frac{dx}{dy}}.$
\item $\displaystyle \frac{d}{dx}(\in v) = \frac{1}{v} \cdot \frac{dv}{dx}.$
\item $\displaystyle \frac{d}{dx}(\log v) = \frac{\log e}{v} \cdot \frac{dv}{dx}.$
\item $\displaystyle \frac{d}{dx}(a^v) = a^v \in a \cdot \frac{dv}{dx}.$
\item $\displaystyle \frac{d}{dx}(e^v) = e^v \cdot \frac{dv}{dx}.$
\item $\displaystyle \frac{d}{dx}(u^v) = u^v \cdot \frac{du}{dx} + \in u \cdot u^v \frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}( \sin v) =  \cos v  \cdot  \frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}( \cos v) =  -\sin v  \cdot  \frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}( \tan v) =  \sec^2 v \cdot  \frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}( \cot v) =  -\csc^2 v \cdot \frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}( \sec v) = \tan v\cdot \frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}( \csc v) = -\csc v \cdotc^frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}(\arcsin v) = \frac{1}{\sqrt{1 - v^2}} \cdot \frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}(\arccos v) = \frac{1}{\sqrt{1 - v^2}} \cdot \frac{dv}{dx}.$
\item $ \displaystyle  \frac{d}{dx}(\arctan v) = \frac{1}{\sqrt{1 + v^2}} \cdot \frac{dv}{dx}.$
    \end{enumerate}
